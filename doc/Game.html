def check_game_over
  unless @player1.is_alive? && @player2.is_alive?
    @game_over = true
    announce_winner
  end

    start_game() click to toggle source

    Starts the game, managing turns until the game is over.

    # File game.rb, line 15
def start_game
  puts "COMMENCE MATHING NOW!"
  current_turn = 1
  until @game_over
    puts "----- TURN #{current_turn} -----"
    turn = Turn.new(@current_player, @player1, @player2)
    turn.execute_turn
    check_game_over
    switch_turns
    current_turn += 1
  end
end
